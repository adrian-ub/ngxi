```sh
npm install @ngxi/quill
```

## Usage

```ts
import { Component } from '@angular/core'
import { QuillCollapseIcon } from '@ngxi/quill'

@Component({
  imports: [QuillCollapseIcon],
  template: `<svg quillCollapseIcon></svg>`,
})
export class UsageIcon {}
```
