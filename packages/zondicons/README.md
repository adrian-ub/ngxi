## Installation

```sh
npm install @ngxi/zondicons
```

## Usage

```ts
import { Component } from '@angular/core'
import { ZondiconsCopyIcon } from '@ngxi/zondicons'

@Component({
  imports: [ZondiconsCopyIcon],
  template: `<svg zondiconsCopyIcon></svg>`,
})
export class UsageIcon {}
```
