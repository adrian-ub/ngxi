## Installation

```sh
npm install @ngxi/zmdi
```

## Usage

```ts
import { Component } from '@angular/core'
import { ZmdiAlarmSnoozeIcon } from '@ngxi/zmdi'

@Component({
  imports: [ZmdiAlarmSnoozeIcon],
  template: `<svg zmdiAlarmSnoozeIcon></svg>`,
})
export class UsageIcon {}
```
