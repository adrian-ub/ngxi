## Installation

```sh
npm install @ngxi/fa-regular
```

## Usage

```ts
import { Component } from '@angular/core'
import { FaRegularBellIcon } from '@ngxi/fa-regular'

@Component({
  imports: [FaRegularBellIcon],
  template: `<svg faRegularBellIcon></svg>`,
})
export class UsageIcon {}
```
