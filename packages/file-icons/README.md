## Installation

```sh
npm install @ngxi/file-icons
```

## Usage

```ts
import { Component } from '@angular/core'
import { FileIconsAdobeIcon } from '@ngxi/file-icons'

@Component({
  imports: [FileIconsAdobeIcon],
  template: `<svg fileIconsAdobeIcon></svg>`,
})
export class UsageIcon {}
```
