```sh
npm install @ngxi/icons8
```

## Usage

```ts
import { Component } from '@angular/core'
import { Icons8CheckedIcon } from '@ngxi/icons8'

@Component({
  imports: [Icons8CheckedIcon],
  template: `<svg icons8CheckedIcon></svg>`,
})
export class UsageIcon {}
```
