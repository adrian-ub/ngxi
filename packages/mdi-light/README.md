## Installation

```sh
npm install @ngxi/mdi-light
```

## Usage

```ts
import { Component } from '@angular/core'
import { MdiLightCartIcon } from '@ngxi/mdi-light'

@Component({
  imports: [MdiLightCartIcon],
  template: `<svg mdiLightCartIcon></svg>`,
})
export class UsageIcon {}
```
